## useTokenExplorer

Hook for navigating through minted tokens.

```tsx
import { useTokenExplorer } from '@public-assembly/builder-utils'
```

## Usage

```tsx
import { useTokenExplorer } from '@public-assembly/builder-utils'

function App() {
  const { tokenId, currentTokenId, incrementId, decrementId, isFirstToken, isLastToken } =
    useTokenExplorer()

  return (
    <>
      <p>Latest minted token: {tokenId}</p>
      <p>Now viewing token: {currentTokenId}</p>
      <>
        <button disabled={isFirstToken} onClick={decrementId}>
          Backward
        </button>
        <button disabled={isLastToken} onClick={incrementId}>
          Forward
        </button>
      </>
    </>
  )
}
```

## Returns

```tsx
{
  tokenId: number
  currentTokenId: number
  incrementId: () => void
  decrementId: () => void
  isFirstToken: boolean
  isLastToken: boolean
}
```
